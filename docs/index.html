The reason why I am making this js & css files, is to help me take notes for my game development study. I am learning Unreal Engine 4, and they are using visual scripting, which means I need to take lot of huge screenshot of the visual script during my study. I was gonna make study notes in pdf, but I didn't think it will work well, and if I just scatter it in jpg it will be a mess...

So my solution is to take study notes in html and make my own js & css that fit my criteria of good study notes.
My preferences for good study notes are...

  • Contain lot information in a page, just like books
  • Small readable font, instead of big fonts
  • Hidable element, cause I don't like long scroll
  • Plus you can embed video in html notes

Since my script that was meant to help me take study notes, getting bigger and bigger, and it's kinda getting messy, I need to make the documentation for the script. So this page will be documentation of ar3.js and ar3.css.

construction phase

Construction phase only will be done once, at the start of web load, and that's it. If people download the page and open the offline version than the construction will not be executed.

  • If F is released then maxWalkSpeed = defaultWalkSpeed and isSprinting = false.

//##############################################
                  // variable outside .tab-r3 functionality      #
                  // to make sure element will not be construct  #
                  // if it is already construct                  #
                  //############################################## initialized
                  var $ar3 = $('body'),
                      newElement = !$ar3.hasClass('no_more_new_element');

                  //##########
                  // .tab-r3 # contain 1st div that will be the links, and 2nd div that will be the content
                  //##########
                  (function() {

                      var $tabGroup = $('.tab-r3'),
                          $div2 = $tabGroup.children('div:nth-child(2)'),
                          $divContent = $div2.children('div'),
                          divContentL = $divContent.length;

                      //###########
                      // <div>2nd # access div content
                      //###########
                      (function() {

                          //########
                          // <div> # Create ID for every div content
                          //########
                          (function() {
                              for (i = 0; i < divContentL; i++) {
                                  $($divContent[i]).attr('id', 'r3_tab_content_'+ i);
                              }
                          })();

                          //######
                          // <a> # attach click function in div content (additional functionality)
                          //######
                          $div2.find('div.relative-container-r3 > a').click(function() {
                              var n = $(this).attr('data-tab-a');
                              $(this).parents('.tab-r3')
                                .children('div:first-child')
                                .children('a:nth-of-type('+ n +')').click();
                          });

                      })();

                      //###########
                      // <div>1st #
                      //###########
                      (function() {
                          var $div1 = $tabGroup.children('div:first-child'),
                              $a = $div1.children('a');

                          //######
                          // <a> # create href attribute to <a>
                          //######
                          (function() {
                              if (newElement) {
                                  for (i = 0; i < divContentL; i++) {
                                      var id = $($divContent[i]).attr('id');
                                      $a[i].setAttribute('href', '#'+id);
                                  }
                              }
                          })();

                          //##################
                          // <a>:first-child # first <a> is activated by default
                          //##################
                          (function() {
                              if (newElement) {
                                  $div1.children('a:first-child').addClass('r3-active');
                              }
                          })();

                          //######
                          // <a> # attach click event
                          //######
                          $a.click(function() {
                              var id = this.href.slice(this.href.indexOf('#') + 1, this.href.length);
                              $(this).siblings().removeClass('r3-active');
                              $(this).addClass('r3-active');
                              $('#'+id).siblings().fadeOut(400);
                              $('#'+id).delay(401).fadeIn();
                          });

                      })();

                  })();

                  //##############################################
                  // variable outside .tab-r3 functionality      #
                  // to make sure element will not be construct  #
                  // if it is already construct                  #
                  //############################################## check
                  $ar3.addClass('no_more_new_element');
